```
# for에서 리스트를 이용
# 학생 성적 리스트 score = [70, 90, 100, 50, 85]
# 60점 이상은 합격, 미만은 불합격 출력
# 1번 학생 : 합격
# 2번 학생 : 합격
...
# 5번 학생 : 합격


score = [70, 90, 100, 50, 85]
count = 0
for i in score:
    count += 1
    if i >= 60:
        print('{}번 학생 {}점 합격'.format(count, i))
    else:
        print('{}번 학생 {}점 불합격'.format(count, i))
# 몇번 학생을 어떻게 해야하지?
# for문 밖에서 초기값을 설정한 후, for문 안에서 for문이 돌아갈 때마다 횟수가 누적될 수 있도록 변수 설정





# 숫자10개 제시하고 양수,음수,0 개수를 카운트하기

pos, neg, zero = 0, 0, 0

for i in range(1, 11):
    number = int(input('숫자' + str(i) + '입력: '))
    if number > 0:
        pos += 1
    elif number == 0:
        zero += 1
    else:
        neg += 1

print('------------')
print('양수 개수 :', pos)
print('음수 개수 :', neg)
print('0의 개수 :', zero)
```

```
# 조건 맞으면 그 다음 반복을 수행

for i in range(10):
    if i % 2 == 0:
        continue
    print(i)


print('------------------')


i = 1
while i < 10:
    if i % 5 == 0:
        break
    print(i)
    i += 1

i = 0

    i += 1
    if i % 2 == 0:
        continue
```
